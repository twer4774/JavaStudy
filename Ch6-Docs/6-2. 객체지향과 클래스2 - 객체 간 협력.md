# 6-2. 객체지향과 클래스2 - 객체 간 협력

- Student, Bus, Subway 객체들의 협력하는 방법
  - 학생은 버스나 지하철을 이용할 수 있음
  - 학생이 버스를 선택하면 학생이 가진 돈은 1000원이 줄고, 지하철을 이용하면 1500원 줄어듦
    - 반대로 버스는 1000원이 증가, 지하철은 1500원 증가

```java
//Sutdent
package ObjectCooperation;

public class Student {
    public String studentName; //학생이름
    public int grade; //학년
    public int money; //학생이 가지고 있는 돈

    public Student(String studentName, int money) {
        this.studentName = studentName;
        this.money = money;
    }

    //버스를 타면 1000원 지출
    public void takeBus(Bus bus) {
        bus.take(1000);
        this.money -= 1000;
    }

    //지하철을 타면 1500원 지출
    public void takeSubway(Subway subay){
        subay.take(1500);
        this.money -= 1500;
    }

    public void showInfo(){
        System.out.println(studentName + "님의 남은 돈은" + money + "입니다.");
    }
}


//Bus
package ObjectCooperation;

public class Bus {

    int busNumber;
    int passengerCount;
    int money;

    public Bus(int busNumber) {
        this.busNumber = busNumber; //버스번호를 매개변수로 받는 생성자
    }

    public void take(int money){
        this.money += money;
        passengerCount++;
    }

    public void showInfo(){
        System.out.println("버스 " + busNumber + "번의 승객은 " + passengerCount + "명이고,  수입은 " + money + "입니다.");
    }
}


//Subway
package ObjectCooperation;

public class Subway {

    String lineNumber;
    int passengerCount;
    int money;

    public Subway(String lineNumber){
        this.lineNumber = lineNumber;
    }

    public void take(int money){
        this.money += money;
        passengerCount++;
    }

    public void showInfo(){
        System.out.println(lineNumber + "의 승객은 " + passengerCount + "명이고, 수입은 " + money + "입니다.");
    }
}

//객체간 협력을 구현한 클래스
package ObjectCooperation;

public class TakeTrans {

    public static void main(String[] args) {
        Student studentJames = new Student("James", 5000);
        Student studentTomas = new Student("Tomas", 10000);

        Bus bus100 = new Bus(100); //노선이 100번 버스 생성
        studentJames.takeBus(bus100);
        studentJames.showInfo();
        bus100.showInfo();

        Subway subwayGreen = new Subway("2호선");
        studentTomas.takeSubway(subwayGreen);
        studentTomas.showInfo();
        subwayGreen.showInfo();
    }
}
```
